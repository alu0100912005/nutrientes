def <=>(other)
    if @proteinas == other.proteinas
        if @glucidos == other.glucidos
            if @grasas == other.grasas
                return 0
                elsif
                    grasas > other.grasas
                    return 1
                else
                    return -1
            end
        elsif
            glucidos > other.glucidos
            return 1
        else
            return -1
        end
    elsif
        proteinas > other.proteinas
        return 1
    else
        return -1
    end
end

def get_ve
    sum = (@proteinas * 4) + (@glucidos * 4) + (@grasas * 9)
    # aux = "#{sum}" + " Kcal"
    return sum
end
